pipeline {
environment {
registry = "abhishek/myimages"
registryCredential = 'docker-build' (- - create docker hub credentials in Jenkins)
dockerImage = ''
}
agent any
stages {
stage('Cloning our Git') {
steps {
git 'https://github.com/your repository'
 }
}
stage('Building our image') {
steps{
script {
dockerImage = docker.build registry + ":$BUILD_NUMBER"
  }
 }
}
stage('Deploy our image') {
steps{
script {
docker.withRegistry('https://registry.hub.docker.com', 'docker-build') {
dockerImage.push()
   }
  }
 }
}
stage('Cleaning up') {
steps{
sh "docker rmi $registry:$BUILD_NUMBER"
   }
  }
 }
}
